class Beam:
    def __init__(self, stringerDict, l=1500, w=400, h=150, t=0.8, stringT = 1.5, rivetSpacing = 20):
        self.l = l
        self.w = w
        self.h = h
        self.t = t
        self.stringerDict = stringerDict
        self.stringT = stringT
        self.s = rivetSpacing
        
    def getStringDib(self, x): #this function gives you the number of stiffeners on top & bottom panels at distance x from root
        keys = list(self.stringerDict.keys())
        nString = 0
        for i in range(len(keys)):
            if x < keys[i]:
                nString = self.stringerDict[keys[i-1]]
                break
        if nString == 0:
            nString = self.stringerDict[keys[-1]]
        return nString
    
    def yNeutral(self, x):
        nString = self.getStringDib(x)
        
        hArea = self.t*self.w
        vArea = self.t*(self.h-self.t)
        aStringUp = 60*nString[0]
        aStringDown = 60*nString[1]
        
        totalArea = 2*hArea+2*vArea+aStringUp+aStringDown
        
        y = (aStringUp*self.h/2 - aStringDown*self.h/2)/totalArea
        return y, totalArea
    
    def MoI(self, x):
        nString = self.getStringDib(x)
        y, _ = self.yNeutral(x)

        MBox = 3.99749E6
#         MBox = 4.05E6
        MoIStringer = 2238.7
    
        hArea = self.t*self.w
        vArea = self.t*(self.h-self.t)
        boxArea = (2*hArea+2*vArea)
        stringerArea = 57.75
        
#         print(boxArea)

        boxMoI = MBox + boxArea * y**2
        upperMoI = nString[0]*(MoIStringer + stringerArea*(self.h/2-y)**2)
        lowerMoI = nString[1]*(MoIStringer + stringerArea*(self.h/2+y)**2)
        
        return boxMoI + upperMoI + lowerMoI 
    
    def getStressDict(self, x):
        stressDict = dict()
        I = self.MoI(x)
        y, _ = self.yNeutral(x)
        
        stringDist = self.getStringDib(x)
        pitchUp = self.w/(stringDist[0]+1)
        pitchDown = self.w/(stringDist[1]+1)
        
        M = tipLoad*(self.l-x)
        stressDict["shear"] =  4/3*0.25 * (2*tipLoad/(self.t*self.h/2))
        stressDict["upperBend"] = 4/3*M * (self.h/2 - y)/I
        stressDict["lowerBend"] = -4/3* M *  (self.h/2 + y)/I
        
        
        
        # Following code not actually used !
        
        Kc = 12 #arbitrary values  for demonstration purposes
        Ks = 8
        
        stressDict["bucklingStressUp"] = Kc * aluSheet["EModulus"] * (self.t/pitchUp)**2
        
        stressDict["bucklingShearUp"] = Ks*aluSheet["EModulus"] * (self.t/pitchUp)**2
        stressDict["bucklingShearDown"] = Ks*aluSheet["EModulus"] * (self.t/pitchDown)**2
    
        stressDict["rivetBuckling"] = 0.9* 2.1 * aluSheet["EModulus"] * (self.t/self.s)**2
        
        return stressDict
