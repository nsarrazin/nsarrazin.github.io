# #green
def func1(array):
    #converts RGB to HLS 
    hls = cv2.cvtColor(array,cv2.COLOR_RGB2HLS)
    
    #count the number of pixel with a hue between 60 and 200 degrees
    truthtable = hls[hls[:,:,0] > 60/2]
    truthtable = truthtable[:,:] < 200/2
    return np.sum(truthtable)/(array.shape[0]*array.shape[1])

#blue
def func2(array):
    #same stuff as func1 except for blue
    hls = cv2.cvtColor(array,cv2.COLOR_RGB2HLS)
    truthtable = hls[hls[:,:,0] > 200/2]
    truthtable = truthtable[:,:] < 240/2
    
    return np.sum(truthtable)/(array.shape[0]*array.shape[1])


x_array = []
